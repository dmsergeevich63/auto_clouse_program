```delphi
uses
  FireDAC.Stan.Intf, FireDAC.Stan.Option, FireDAC.Stan.Error, FireDAC.UI.Intf,
  FireDAC.Phys.PGDef, FireDAC.Phys.PG, FireDAC.Stan.Def, FireDAC.Stan.Pool,
  FireDAC.Stan.Async, FireDAC.VCLUI.Wait, FireDAC.Phys, FireDAC.Comp.Client,
  Data.DB;

var
  FDConnection: TFDConnection;
  FDQuery: TFDQuery;
begin
  FDConnection := TFDConnection.Create(nil);
  FDQuery := TFDQuery.Create(nil);
  try
    FDConnection.Params.Add('DriverID=PG');
    FDConnection.Params.Add('Database=YourDatabase');
    FDConnection.Params.Add('User_Name=YourUserName');
    FDConnection.Params.Add('Password=YourPassword');
    FDConnection.Open;

    FDQuery.Connection := FDConnection;
    FDQuery.SQL.Text :=
      'DELETE FROM YourTable ' +
      'WHERE YourField !~ ''^.$''';  // Удаляем строки, содержащие неверные последовательности байт для UTF-8

    FDQuery.ExecSQL;
  finally
    FDQuery.Free;
    FDConnection.Free;
  end;
end;
```

В этом примере мы использовали компоненты FireDAC для установки соединения с базой данных PostgreSQL и выполнения SQL-запроса для удаления данных, содержащих неверные последовательности байт для UTF-8 из таблицы `YourTable` в базе данных.

Убедитесь, что вы установили и сконфигурировали FireDAC для работы с PostgreSQL, и замените `YourDatabase`, `YourUserName`, `YourPassword` и `YourTable` на реальные значения вашей базы данных и таблицы.
